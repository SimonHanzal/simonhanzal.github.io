In the following code, I use a combination of base R and Tidyverse functions to obtain indeces of trials since participant commission error (post-error slowing) and then programme and apply a custom summary function.

# Adding column of time since error
df_switching_since <- df_switching %>%
    arrange(participant, trial) %>%
    # A temporary column marks an accurance of a mistake
    group_by(participant, temp = rev(cumsum(rev(Stimulus.ACC==0)))) %>%
    # With following entries in ascending order
    mutate(since = ifelse(Stimulus.ACC==0, n()-1, NA)) %>%
    ungroup() %>%
    select(-temp)

# A base r loop to reset the index with each mistake
no = 1
repeat {
    no = no + 1  
    df_switching_since$since = ifelse(!is.na(df_switching_since$since),
    df_switching_since$since, lead(df_switching_since$since)-1)
    if(no > (trial_number)) {
        break
    }
}

# Finishing touches
df_switching_since <- df_switching_since %>%
    mutate(since = ifelse(since < 0, 0, since))

# A summary function summarising a metric at a
# specific index of post-error slowing
summarise_error_df <- function(df, since_index = since,
    summary_metric = mean_rt, position = 1) {
  result <- df %>%
    # Using indirection to treat NSE
    group_by({{since_index}}) %>%
    summarise(mean = mean({summary_metric},
              sd = sd({{summary_metric}}))) %>%
    filter({{since_index}} == {{position}}) %>%
    deframe()
  return(result)
}

# Applying the function
df_switching_summary <- summarise_error_df(df_switching_since)

# A matplotlib frame redraw function
def animate(self, i, xc: list, yc: list):
    # Pogresses the animation slightly ahead of the sampling rate
    if (self.counter - len(tracked_signal)) < -80:
        change = 80
    # Readjust the change score to account for any delays
    else: 
        change = len(tracked_signal) - self.previous_signal
    # Update x and y values
    self.counter += change
    self.addition = tracked_signal[self.counter - change:self.counter]
    # To maximise performance, plot axis are adjusted reactively
    if max(self.addition) > self.axlim:
        self.axlim = max(self.addition)
    elif max(self.addition) < -self.axlim:
        self.axlim = abs(max(self.addition1))
    # Add x and y to lists
    self.xc.extend(list(range(self.counter-change+1, self.counter+1)))
    self.yc.extend(self.addition1)
    # Limit x and y lists to constant sampling rate to show one second
    xc = self.xc[-SRATE:]
    yc = self.yc[-SRATE:]
    # Draw x and y lists
    self.ax.clear()
    self.ax.plot(xc, yc)
    # Format plot
    self.ax.set_ylim([-self.axlim, self.axlim])
    plt.xticks(rotation=45, ha='right')
    plt.subplots_adjust(bottom=0.20)
    self.ax.set_title('Plot of server signal')
    self.ax.set_xlabel('Sample no')
    self.ax.set_ylabel('mV')
    # Log the length of the signal at this point
    self.previous_signal = len(tracked_signal)
    # Exiting behaviour
    if self.view_complete:
        self.ani.event_source.stop()

# An animation triggered by outside events within a PyQt widget
def display(self):
    # Initialising function variables
    global tracked_signal
    self.counter = 0
    self.axlim = 0.1
    self.fig, self.ax = plt.subplots()
    self.xc = []
    self.yc = []
    # Starts a lagging-behind tracker
    self.lag_tracker = threading.Thread(target=self.do_check)
    self.lag_tracker.start()
    # Runs the animation
    self.ani = animation.FuncAnimation(
        self.fig, self.animate, fargs=(self.xc, self.yc), interval=40)
    plt.show()
